// --- タイマー・リセット機能 --- let timerInterval = null; let timerValue = 0; let timerRunning = false; const timerBtn = document.getElementById('timer-btn'); const timerDisplay = document.getElementById('timer-display'); const resetBtn = document.getElementById('reset-btn'); // 読み上げ用 function speak(text) { if (window.speechSynthesis) { const uttr = new window.SpeechSynthesisUtterance(text); uttr.lang = 'ja-JP'; window.speechSynthesis.speak(uttr); } } function startTimer() { if (timerRunning) return; timerValue = 30; timerDisplay.textContent = timerValue + '秒'; timerRunning = true; timerBtn.disabled = true; // タイマー中はリセットボタンを無効化 resetBtn.disabled = true; let alreadySpoken = {}; timerInterval = setInterval(() => { timerValue--; timerDisplay.textContent = timerValue > 0 ? timerValue + '秒' : '0秒'; if (timerValue <= 5 && timerValue > 0 && !alreadySpoken[timerValue]) { speak(timerValue + ''); alreadySpoken[timerValue] = true; } if (timerValue <= 0) { clearInterval(timerInterval); timerRunning = false; timerBtn.disabled = false; resetBtn.disabled = false; timerDisplay.textContent = '終了'; // 画面上の魚を高速で逃がす escapeAllSakana(); } }, 1000); } function resetTimer() { if (timerInterval) clearInterval(timerInterval); timerValue = 0; timerDisplay.textContent = ''; timerRunning = false; timerBtn.disabled = false; resetBtn.disabled = false; // 魚もリセット removeAllSakana(); } timerBtn.addEventListener('click', startTimer); resetBtn.addEventListener('click', resetTimer); // 魚をすべて消す（リセット用） function removeAllSakana() { document.querySelectorAll('.sakana').forEach(sakana => { sakana.remove(); }); sakanaCount = 0; } // 魚を高速で逃がす（タイマー終了時） function escapeAllSakana() { document.querySelectorAll('.sakana').forEach(sakana => { // すでに逃走中なら何もしない if (sakana._escaping) return; sakana._escaping = true; // 逃げる方向を決める（今の向きのまま or 画面外へ最短） const rect = sakana.getBoundingClientRect(); const centerX = rect.left + rect.width/2; let direction = (centerX < window.innerWidth/2) ? -1 : 1; // 右端なら右へ、左端なら左へ let speed = 30 + Math.random()*20; function escapeMove() { let x = parseFloat(sakana.style.left); x += direction * speed; sakana.style.left = x + 'px'; if ((direction === 1 && x > window.innerWidth + 200) || (direction === -1 && x < -200)) { sakana.remove(); sakanaCount--; } else { requestAnimationFrame(escapeMove); } } escapeMove(); }); }
